# ============================================================================
# MARTA-INSIGHTS cron block — the jobs that work TODAY.
# ============================================================================
#
# Install with:   scripts/marta/install-crontab.sh
# (safe marker-merge — replaces only this block, never your other cron jobs.)
#
# Data-capture, schedule-refresh, and the first detect→post job (bus bunching)
# run here; remaining detector/posting/export jobs get added as they land.
# __REPO__ and __NODE__ are filled in at install.
# Logs go to __REPO__/state/logs/ (gitignored).

# MARTA-INSIGHTS-START
# Bus positions: every minute, 2 ticks 30s apart → 30s density (feeds every bus
# detector). Cheap (~160 rows/fetch).
* * * * * cd __REPO__ && __NODE__ scripts/marta/observe-buses.js >> __REPO__/state/logs/observe-buses.log 2>&1
# Rail (trains + arrivals): every minute, 2 ticks 30s apart for position-delta speed.
* * * * * cd __REPO__ && __NODE__ scripts/marta/observe-rail.js >> __REPO__/state/logs/observe-rail.log 2>&1
# Bus TripUpdates: every 5 min. Stores compact trip-level status, including
# MARTA-announced cancellations used to enrich ghost-bus detection. Full
# stop-level prediction rows are opt-in only (MARTA_STORE_TRIP_UPDATE_STOPS=1).
*/5 * * * * cd __REPO__ && __NODE__ scripts/marta/observe-bus-tripupdates.js >> __REPO__/state/logs/observe-bus-tripupdates.log 2>&1
# Refresh static GTFS + rebuild the schedule index nightly at 03:30.
30 3 * * * cd __REPO__ && __NODE__ scripts/marta/fetch-static-gtfs.js && __NODE__ scripts/marta/build-schedule-index.js >> __REPO__/state/logs/gtfs-refresh.log 2>&1
# Bus bunching detect→post (martabusinsights). Every 5 min, offset off the observe
# minutes; reads the latest snapshot the observe loop recorded (no extra fetch).
# Posting volume is bounded by per-shape+route 1h cooldowns and a 3/route/day cap.
# Cross-route bus pileup (martabusinsights). Every 5 min, 1 min BEFORE the
# per-route bunching job so a posted multi-route pileup suppresses the per-route
# post for the same buses (recentCrossBunchMemberIds, 10-min window).
1-59/5 * * * * cd __REPO__ && __NODE__ bin/marta/bus/cross-bunching.js >> __REPO__/state/logs/bus-cross-bunching.log 2>&1
2-59/5 * * * * cd __REPO__ && __NODE__ bin/marta/bus/bunching.js >> __REPO__/state/logs/bus-bunching.log 2>&1
# Bus gap detect→post (martabusinsights). Every 5 min, offset from bunching.
# Reads the latest observed snapshot; bounded by per-shape+route cooldowns and
# a 3/route/day cap with severity escalation overrides.
4-59/5 * * * * cd __REPO__ && __NODE__ bin/marta/bus/gaps.js >> __REPO__/state/logs/bus-gaps.log 2>&1
# Bus ghost rollup (martabusinsights). Every 15 min; uses a 60-min observation window.
7-59/15 * * * * cd __REPO__ && __NODE__ bin/marta/bus/ghosts.js >> __REPO__/state/logs/bus-ghosts.log 2>&1
# Bus cancellation surge → roundup signals (signal-only, no post). Every 15 min on a
# 60-min window; lands just before the incident-roundup tick (3-59/5) reads them.
6-59/15 * * * * cd __REPO__ && __NODE__ bin/marta/bus/cancellation-surge.js >> __REPO__/state/logs/bus-cancellation-surge.log 2>&1
# Bus speedmap (martabusinsights). Hourly route rotation from the last hour of observed speeds.
12 * * * * cd __REPO__ && __NODE__ bin/marta/bus/speedmap.js >> __REPO__/state/logs/bus-speedmap.log 2>&1
# Bus thin-service gaps (martabusinsights). Every 15 min; low-frequency routes
# (headway >=20 min) the mainline gap/ghost detectors are structurally blind to.
# One post/route/day; threads a clear reply when buses reappear.
13-58/15 * * * * cd __REPO__ && __NODE__ bin/marta/bus/thin-gaps.js >> __REPO__/state/logs/bus-thin-gaps.log 2>&1
# Bus pulse / route blackouts (martabusinsights). Every 5 min; higher-frequency
# routes (headway <20 min) showing zero vehicles while the fleet reports normally.
14-59/5 * * * * cd __REPO__ && __NODE__ bin/marta/bus/pulse.js >> __REPO__/state/logs/bus-pulse.log 2>&1
# Rail gap and bunching detect→post (martatraininsights). Offset from bus jobs.
6-59/5 * * * * cd __REPO__ && __NODE__ bin/marta/rail/gaps.js >> __REPO__/state/logs/rail-gaps.log 2>&1
# Cross-line rail pileup (martatraininsights). 1 min BEFORE the per-line bunching
# job so a posted multi-line pileup (e.g. RED+GOLD at Five Points) suppresses the
# per-line post for the same trains.
7-59/5 * * * * cd __REPO__ && __NODE__ bin/marta/rail/cross-bunching.js >> __REPO__/state/logs/rail-cross-bunching.log 2>&1
8-59/5 * * * * cd __REPO__ && __NODE__ bin/marta/rail/bunching.js >> __REPO__/state/logs/rail-bunching.log 2>&1
# Rail ghost rollup and speedmap rotation (martatraininsights).
10-59/15 * * * * cd __REPO__ && __NODE__ bin/marta/rail/ghosts.js >> __REPO__/state/logs/rail-ghosts.log 2>&1
17 * * * * cd __REPO__ && __NODE__ bin/marta/rail/speedmap.js >> __REPO__/state/logs/rail-speedmap.log 2>&1
# Rail system timelapse (martatraininsights). Every 3 hours; reads the last 15 min
# of stored rail observations (no extra fetch) and posts a smooth, interpolated
# system-wide clip of every train in service.
25 */3 * * * cd __REPO__ && __NODE__ bin/marta/rail/timelapse.js >> __REPO__/state/logs/rail-timelapse.log 2>&1
# Rail dead-segment "pulse" (martaalertinsights). Every 2 min (even minutes, offset
# from the official-alerts job). Flags a stretch of track between stations that no
# train has passed through for longer than the schedule allows (>= max(2.5x headway,
# 15 min)); posts after 3 consecutive ticks, threads a clear reply after 5 clean
# ticks, and threads under any open official rail alert. Reads the latest stored
# rail observations (no extra fetch).
0-58/2 * * * * cd __REPO__ && __NODE__ bin/marta/rail/pulse.js >> __REPO__/state/logs/rail-pulse.log 2>&1
# Official service alerts (martaalertinsights). Every 2 min (odd minutes, offset
# from the insight jobs); republishes significant GTFS-rt ServiceAlerts and posts
# a threaded "resolved" reply when one clears. The 2-min cadence pairs with
# ALERT_CLEAR_TICKS=3 for a ~6-min flicker-safe clear window (src/marta/alert/store.js).
1-59/2 * * * * cd __REPO__ && __NODE__ bin/marta/alerts.js >> __REPO__/state/logs/alerts.log 2>&1
# Bot-detected incident roundup (martaalertinsights). Mirrors CTA's alerts bot:
# reads detector meta_signals and posts correlated degraded-service issues from
# the alerts account, with resolution replies only from that same account.
3-59/5 * * * * cd __REPO__ && __NODE__ bin/marta/incident-roundup.js >> __REPO__/state/logs/incident-roundup.log 2>&1
# Bus cancellation rollup (martaalertinsights). Hourly per-route digest of GTFS-rt
# TripUpdates CANCELED trips (same source the ghost detector subtracts), deduped
# in a ledger and posted once, fire-and-forget. NOT the per-route OTP cancellation
# prose (a different measurement — see src/marta/bus/cancellations.js). Off the :00
# minute and away from the other alert jobs.
7 * * * * cd __REPO__ && __NODE__ bin/marta/bus/cancellations.js >> __REPO__/state/logs/bus-cancellations.log 2>&1
# Publish web data to R2 + trigger a site rebuild. Incident/detector posts kick
# this immediately via the webPushTrigger flush; this every-15-min run is the
# backstop that catches any change that didn't ride a post (and re-publishes if
# an upload was missed). Exits early when the exported data is unchanged.
# NODE is passed explicitly because cron's PATH usually lacks node.
14-59/15 * * * * cd __REPO__ && NODE=__NODE__ /bin/sh __REPO__/bin/marta/push-web-data.sh >> __REPO__/state/logs/push-web-data.log 2>&1
# Off-box backup of state/marta.sqlite to Cloudflare R2. Uses SQLite .backup,
# integrity-checks the snapshot, gzips it, then uploads to the private backup
# bucket. See docs/MARTA_BACKUPS.md.
23 4 * * * cd __REPO__ && /bin/bash scripts/marta/backup-db.sh >> __REPO__/state/logs/backup-db.log 2>&1
# MARTA-INSIGHTS-END
